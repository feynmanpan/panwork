url_headers = {'user-agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/63.0.3239.132 Safari/537.36'}
#
begin=9789571300000 #13碼ISBN，時報=13
end=  9789571399999 
batch=10 #每次筆數
epoch=int(((end-begin+1)/batch)/1) #10000 #搜尋次數
pgids={}
#
startT = time()
#______________________________________________
for i in range(5870,epoch):
    #
    begin_pgid=begin+i*batch
    query=str(begin_pgid)
    for j in range(batch-1):
        query+='+'+str(begin_pgid+j+1)
    #    
    clear_output(wait=True)
    print(i,begin_pgid)
    #先搜尋
    url='http://www.eslite.com/Search_BW.aspx?query='+query+'&searchType='
    r = requests.get(url, headers=url_headers)
    r.encoding='utf8'
    result=BeautifulSoup(r.text,"lxml")
    lists=result.select("#ctl00_ContentPlaceHolder1_lbTotalResultCount")
    n=lists[0].string   
    if n=='0':
        continue
    #至少一筆就抓pgid
    imgs=result.select('img.cover_img')
    for img in imgs:
        #抓pgid
        a=img.parent
        href=a["href"]
        find='pgid=([0-9]{16})&'
        m=re.search(find,href)
        pgid=m.group(1)
        #抓書名
        name=img['alt']        
        #抓出版社
        tr=a.parent.parent.find_next_sibling()
        a2=tr.select("span[id$=LblManufacturerName] a")
        publisher=a2[0].string
        #抓出版日期
        span_date=tr.select("span[id$=LblManufacturerDate]")
        publish_date=span_date[0].text.lstrip('出版日期:')
        #抓折扣&價錢
        span_off=tr.select("span.price_sale")
        if '元' in span_off[1].text: #沒有折扣
            count_off=None
            price_off=int(span_off[1].text.replace('元','').replace(',',''))
            price=price_off
        else:            
            count_off=(len(span_off[1].text)<2 and int(span_off[1].text)/10) or int(span_off[1].text)/100
            price_off=int(span_off[2].text.replace('元','').replace(',',''))
            price=round(price_off/count_off)
        #
        pgids[pgid]=[name,publisher,publish_date,str(count_off),str(price_off),str(price)]
        #print(pgids[pgid])
    #
#______________________________________________       
endT = time()     
print(f'共搜尋{epoch*batch}組ISBN = {endT - startT} 秒')
print(len(pgids),'組有效pgid')
