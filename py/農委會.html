# 兩點之間的距離________________________
def dis(x, y, kx, ky):
    return int(((kx-x)**2 + (ky-y)**2)**0.5)

# 對每點進行分群________________________
def cluster(xs, ys, cxs, cys):
    #team = [[]]*seed_num
    team = [[] for _ in range(seed_num)]
    #
    for i in range(xs.count()):
        distant = [ dis(xs[i], ys[i], cxs[j], cys[j]) for j in range(seed_num) ]
        flag=distant.index(min(distant))
        team[flag].append([xs[i], ys[i]])
    #
    return team

# 對分群完的元素找出新的群集中心________________________
def re_seed(team, cxs, cys):
    new_seed = []
    #    
    for idx, nodes in enumerate(team):
        sumx = 0
        sumy = 0        
        if nodes:
            tmp=list(zip(*nodes))
            sumx=sum(tmp[0])
            sumy=sum(tmp[1])
            new_seed.append([int(sumx/len(nodes)), int(sumy/len(nodes))])
        else:  
            new_seed.append([cxs[idx], cys[idx]])
    #
    tmp=list(zip(*new_seed))
    ncxs=np.array(tmp[0])
    ncys=np.array(tmp[1])
    return ncxs, ncys

# k-means 分群________________________
def kmeans(xs, ys, cxs, cys, fig):
    team = cluster(xs, ys, cxs, cys)
    ncxs, ncys = re_seed(team, cxs, cys)

    # 判斷群集中心是否不再更動
    if ((ncxs == cxs).all() and (ncys == cys).all()) or fig==1000:    
        print('fig',fig)        
        # 點連線到新群集中心
        line = plt.gca()
        for idx, nodes in enumerate(team):
            for node in nodes:
                line.plot([node[0], ncxs[idx]], [node[1], ncys[idx]], color='r', alpha=0.4)

        # 散點繪圖
        feature   = plt.scatter(xs, ys)
        k_feature = plt.scatter(cxs, cys)
        nk_feaure = plt.scatter(ncxs, ncys)
        #plt.savefig('/yourPATH/kmeans_%s.png' % fig)
        plt.xlim([0,500])
        plt.rc('figure',figsize=(10,10))
        plt.show()   
    else:        
        fig += 1        
        team,ncxs,ncys=kmeans(xs, ys, ncxs, ncys, fig)
    #    
    return team,ncxs,ncys
